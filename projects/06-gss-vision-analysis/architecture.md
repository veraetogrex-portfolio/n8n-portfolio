### Table Specifications

#### `jobs`
```sql
CREATE TABLE jobs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id UUID NOT NULL UNIQUE,
  status TEXT NOT NULL CHECK (status IN (
    'QUEUED', 'PROCESSING', 'CLASSIFIED', 'FAILED', 'MANUAL_REVIEW'
  )),
  created_at TIMESTAMP DEFAULT now(),
  updated_at TIMESTAMP DEFAULT now()
);

-- Statuses:
-- QUEUED: Just created
-- PROCESSING: Vision API is running
-- CLASSIFIED: Successfully analyzed
-- FAILED: Error during first attempt
-- MANUAL_REVIEW: 3 attempts failed, needs human review
```

#### `renders`
```sql
CREATE TABLE renders (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  image_url TEXT NOT NULL,
  status TEXT NOT NULL CHECK (status IN (
    'PENDING', 'CLASSIFIED', 'FAILED'
  )),
  technical_tags JSONB DEFAULT '[]'::jsonb,
  full_analysis JSONB,
  created_at TIMESTAMP DEFAULT now(),
  
  -- Example technical_tags: ["kitchen", "modern", "island", "marble"]
  -- Example full_analysis:
  -- {
  --   "room_type": "kitchen",
  --   "style": "contemporary",
  --   "furniture_count": 8,
  --   "confidence": 0.92
  -- }
);

-- JSONB Queries:
SELECT * FROM renders WHERE technical_tags @> '["kitchen"]';
SELECT * FROM renders WHERE full_analysis ->> 'style' = 'contemporary';
```

#### `generation_attempts`
```sql
CREATE TABLE generation_attempts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  attempt_number INT NOT NULL CHECK (attempt_number BETWEEN 1 AND 3),
  parameters JSONB DEFAULT '{}'::jsonb,
  result TEXT NOT NULL CHECK (result IN (
    'SUCCESS', 'RETRY', 'FATAL_ERROR'
  )),
  error_message TEXT,
  created_at TIMESTAMP DEFAULT now()
);

-- Example parameters:
-- {
--   "steps": 50,
--   "sampler": "DPM++",
--   "cfg_scale": 7.5,
--   "structure_scale": 0.50
-- }
```

### Views for Monitoring

```sql
-- Jobs statistics
CREATE VIEW v_jobs_stats AS
SELECT 
  status,
  COUNT(*) as count,
  ROUND(100.0 * COUNT(*) / (SELECT COUNT(*) FROM jobs), 2) as percentage
FROM jobs
GROUP BY status;

-- Recent renders
CREATE VIEW v_recent_renders AS
SELECT 
  r.id, r.job_id, r.status, r.technical_tags,
  j.property_id, j.status as job_status,
  r.created_at
FROM renders r
JOIN jobs j ON r.job_id = j.id
ORDER BY r.created_at DESC
LIMIT 50;

-- Retry statistics
CREATE VIEW v_retry_stats AS
SELECT 
  job_id,
  COUNT(*) as total_attempts,
  COUNT(CASE WHEN result = 'SUCCESS' THEN 1 END) as successful,
  COUNT(CASE WHEN result = 'FATAL_ERROR' THEN 1 END) as fatal
FROM generation_attempts
GROUP BY job_id;
```

## ğŸ“ˆ Monitoring & Observability

### Key Metrics

```sql
-- Success rate
SELECT 
  status, 
  COUNT(*) as count,
  ROUND(100.0 * COUNT(*) / (SELECT COUNT(*) FROM jobs), 2) as percentage
FROM jobs
WHERE created_at > now() - interval '24 hours'
GROUP BY status;

-- Average attempts before success
SELECT 
  AVG(max_attempt) as avg_attempts
FROM (
  SELECT job_id, MAX(attempt_number) as max_attempt
  FROM generation_attempts
  WHERE result = 'SUCCESS'
  GROUP BY job_id
);

-- Time to completion
SELECT 
  AVG(EXTRACT(EPOCH FROM (updated_at - created_at))) as avg_seconds
FROM jobs
WHERE status IN ('CLASSIFIED', 'FAILED')
  AND created_at > now() - interval '24 hours';
```

### Alerts

- Job status = MANUAL_REVIEW â†’ Slack alert
- API error rate > 10% â†’ System alert
- DB query > 5s â†’ Performance alert
- Webhook timeout â†’ Retry queue
